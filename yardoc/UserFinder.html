def condition_sql
  <<-SQL
    (
        LOWER(first_name) LIKE :search_term
      OR
        LOWER(last_name) LIKE :search_term
      OR
        LOWER(username) LIKE :search_term
      OR
        LOWER(CONCAT(first_name, last_name)) LIKE :search_term
      OR
        LOWER(email) LIKE :search_term
    )
  SQL
end
