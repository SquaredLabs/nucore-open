def set_journal_mask
  f = Facility.all.order(journal_mask: :desc).first
  self.journal_mask = if f && f.journal_mask.match(/^C(\d{2})$/)
                        sprintf("C%02d", Regexp.last_match(1).to_i + 1)
                      else
                        "C01"
                      end
end
