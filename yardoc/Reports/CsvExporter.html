def csv_body

def format_row(row)
  report_hash.values.map do |callable|
    result = if callable.is_a?(Symbol)
               row.public_send(callable)
             else
               callable.call(row)
    end

    if result.is_a?(DateTime)
      format_usa_datetime(result)
    else
      result
    end
  end
rescue => e
  ["*** ERROR WHEN REPORTING ON #{row.class} #{row.id}: #{e.message} ***"]
end
