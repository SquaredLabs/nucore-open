def build_report
  @data = {}
  @totals = DataRow.new
  @reservations.each do |reservation|
    data_value = yield reservation
    @data[reservation.product.name] ||= DataRow.new
    @data[reservation.product.name].add(data_value)
    @totals.add(data_value)
  end
end
